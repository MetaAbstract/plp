/**
 * Возвращает список тем с задержкой Promise
 * @function getSubjects
 * @returns {Subject[]} - массив диалогов
 */
export const getSubjects = () => new Promise((resolve, reject) => {
  const delay = parseInt(Math.random() * 1000)

  setTimeout(() => {
    const subjects = history.map(
      (el) => { return { id: el.id, subject: el.subject, created: el.created } }
    )
    resolve(subjects)
  }, delay)
})
